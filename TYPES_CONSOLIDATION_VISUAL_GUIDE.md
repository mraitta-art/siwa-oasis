### Consolidation Operation

```
SELECT "hotel" AS parent_type
MERGE ["hotel_old", "hotel_legacy"]

UPDATE TYPE HIERARCHY:
  hotel_old.parent_id = "hotel"
  hotel_old.is_parent = FALSE

  hotel_legacy.parent_id = "hotel"
  hotel_legacy.is_parent = FALSE

UPDATE BUSINESSES (12 rows):
  type_id: "hotel_old" вҶ’ "hotel"

UPDATE BUSINESSES (8 rows):
  type_id: "hotel_legacy" вҶ’ "hotel"

UPDATE FORM_FIELDS:
  business_type_id: "hotel_old" вҶ’ "hotel"
  business_type_id: "hotel_legacy" вҶ’ "hotel"

UPDATE CARD_TEMPLATES:
  business_type_id: "hotel_old" вҶ’ "hotel"
  business_type_id: "hotel_legacy" вҶ’ "hotel"

вң… Consolidation Complete
```
